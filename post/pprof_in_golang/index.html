package main

import (
    "fmt"
    "net/http"
    _ "net/http/pprof"
    "time"
)
func main() {
    go func() {
        ticker := time.NewTicker(time.Second)
        for {
            select {
            case t := <- ticker.C:
                fmt.Println("Current time:", t)
            default:
            }
        }
    }()
    http.ListenAndServe(":8080", nil)
}
